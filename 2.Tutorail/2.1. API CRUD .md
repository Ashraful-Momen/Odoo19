# API CRUD

---

# 1. GET ALL Students

```python
@http.route(
    "/api/students",
    type="json",
    auth="public",
    methods=["GET"]
)
def get_students(self):

    students = request.env[
        "student.student"
    ].sudo().search([])

    data = []

    for student in students:

        data.append({

            "id":student.id,
            "name":student.name,
            "age":student.age,
            "email":student.email

        })

    return {

        "status":200,
        "data":data

    }
```

---

# 2. GET Single Student

```python
@http.route(
    "/api/students/<int:student_id>",
    type="json",
    auth="public",
    methods=["GET"]
)
def get_student(self,student_id):

    student = request.env[
        "student.student"
    ].sudo().browse(student_id)

    if not student.exists():

        return {

            "status":404,
            "message":"Student Not Found"

        }

    return {

        "status":200,

        "data":{

            "id":student.id,
            "name":student.name,
            "age":student.age,
            "email":student.email

        }

    }
```

---

# 3. POST Create Student

```python
@http.route(
    "/api/students",
    type="json",
    auth="public",
    methods=["POST"]
)
def create_student(self):

    data = json.loads(
        request.httprequest.data
    )

    student = request.env[
        "student.student"
    ].sudo().create({

        "name":data.get("name"),
        "age":data.get("age"),
        "email":data.get("email")

    })

    return {

        "status":201,

        "id":student.id

    }
```

---

# 4. PUT Update Student

```python
@http.route(
    "/api/students/<int:student_id>",
    type="json",
    auth="public",
    methods=["PUT"]
)
def update_student(self,student_id):

    student = request.env[
        "student.student"
    ].sudo().browse(student_id)

    if not student.exists():

        return {

            "status":404

        }

    data = json.loads(
        request.httprequest.data
    )

    student.write(data)

    return {

        "status":200

    }
```

---

# 5. DELETE Student

```python
@http.route(
    "/api/students/<int:student_id>",
    type="json",
    auth="public",
    methods=["DELETE"]
)
def delete_student(self,student_id):

    student = request.env[
        "student.student"
    ].sudo().browse(student_id)

    if not student.exists():

        return {

            "status":404

        }

    student.unlink()

    return {

        "status":200

    }
```
